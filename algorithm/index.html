# 算法

要求:

  • 正确性：无语法错误、特殊输入正确输出
  • 可读性
  • 健壮性： 输入数据不合法时，可合理处理；
  • 时间效率高/存储量低

算法效率

  • 函数渐进增长（计算次数）：f(n) = n vs n^2
  • 时间复杂度：O(1) O(n) O(n^2) O(logn) 常数阶 线性阶 平方阶 对数阶

# 排序算法

名称 简述 时间复杂度
冒泡排序 往上冒泡，符合条件就两两交换； O(n^2)
选择排序 除数组最后一位，剩下的里面有找大的，两两交换； O(n^2)
插入排序 有序数组（默认第一个有序），后续依次循环，找合适位置插入 O(n^2)
快速排序 数组中间，左小右大，浮标移动至交叉，递归 O(n^logn)
归并排序 拆分要原子维度比较，再合并排序
堆排序 完全二叉树，根节点比左和右都大，创建最大堆，最二大堆...
希尔排序 进阶版插入排序，gap逐渐缩小，知道贴合 O(n^2

冒泡排序

function bubbleSort(arr) {
    if (!Array.isArray(arr) || !arr.length) {
        return arr;
    }
    let len = arr.length;
    // 外层大循环
    for (let i = 0; i < len; i++) {
        // flag：外层当前值；内层小循环
        for (let j = i + 1; j < len; j++) {
            if (arr[i] > arr[j]) {
                let temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
    return arr;
}
let res = bubbleSort([1, 44, 5, 66, 8, 99]);

选择排序

function chooseSort(arr) {
    if (!Array.isArray(arr) || !arr.length || arr.length == 1) {
        return arr;
    }
    let len = arr.length;

    for (let i = len - 1; i > 0; i--) {
        // 除最后一位，找最大值
      	let tempMax = null;
        let tempMaxIndex = null;
 		for (let j = 0; j < i; j++) {
            if (arr[j] > tempMax) {
                tempMax = arr[j];
                tempMaxIndex = j;
            }
        }
        // 满足条件则交换
      	if (tempMax > arr[i]) {
        	let temp = arr[i];
          	arr[i] = tempMax;
          	arr[tempMaxIndex] = temp;
        }
    }
    return arr;
}

let res = chooseSort([1, 4, 5, 66, 8, 99]);

插入排序

function insertSort(arr) {
    if (!Array.isArray(arr) || !arr.length || arr.length == 1) {
        return arr;
    }
    let len = arr.length;

    for (let i = 1; i < len; i++) { // i = 3
        // 0  1  2 | 3
        // 0  1  j | i
        // 1 12 13 | 8
 		let temp = arr[i];
      	let j = i - 1;
        // step1: 寻找位置 + 挪位置
        // 过程demo：1 12 13 | 13  => 1 12 12 | 13 => j=0
      	while(j >= 0 && arr[j] > temp) { // 比flag大
            arr[j + 1] = arr[j]; // 往后挪，cp覆盖 
            j--;
        }
        // 找到位置，插入即可
      	arr[j + 1] = temp;
    }
    return arr;
}

let res = insertSort([100, 4, 54, 66, 8, 99]);
console.log(res);

快速排序

参考：https://humanwhocodes.com/blog/2012/11/27/computer-science-in-javascript-quicksort/

// 交换值
function swap(items, firstIndex, secondIndex){
    var temp = items[firstIndex];
    items[firstIndex] = items[secondIndex];
    items[secondIndex] = temp;
}

function partition(items, left, right) {
    // 中间值 浮标
    var pivot = items[Math.floor((right + left) / 2)],
        i     = left,
        j     = right;

    while (i <= j) {
        // 左标浮动
        while (items[i] < pivot) {
            i++;
        }
        // 右标浮动
        while (items[j] > pivot) {
            j--;
        }
        // 未交叉
        if (i <= j) {
            swap(items, i, j);
            i++;
            j--;
        }
    }
    return i;
}

function quickSort(items, left, right) {
    let index;
    if (items.length > 1) {
        index = partition(items, left, right);
        // 左侧递归
        if (left < index - 1) {
            quickSort(items, left, index - 1);
        }
        // 右侧递归
        if (index < right) {
            quickSort(items, index, right);
        }
    }
    return items;
}


let arr = [100, 4, 54, 66, 8, 99];
let res = quickSort(arr, 0, arr.length - 1);
console.log(res);

归并排序

参考: https://medium.com/techtrument/implementing-merge-sort-in-javascript-898d5f54a234

function merge(leftArr, rightArr) {
    var sortedArr = [];
    while (leftArr.length && rightArr.length) {
        if (leftArr[0] <= rightArr[0]) {
            sortedArr.push(leftArr[0]);
            leftArr = leftArr.slice(1)
        } else {
            sortedArr.push(rightArr[0]);
            rightArr = rightArr.slice(1)
        }
    }
    // 上面基本把小的塞进去了，大的遗调的通过如下操作回归；
    while (leftArr.length) sortedArr.push(leftArr.shift());
    while (rightArr.length) sortedArr.push(rightArr.shift());
    return sortedArr;
}
function mergeSort(arr) {
    // 最终进入merge方法内的数组长度为 1
    if (arr.length < 2) {
        return arr;
    }
    var midpoint = parseInt(arr.length / 2);
    var leftArr  = arr.slice(0, midpoint);
    var rightArr = arr.slice(midpoint, arr.length);
    return merge(mergeSort(leftArr), mergeSort(rightArr));
}

var unsortedArr = [10, -1, 2, 5, 0, 6, 4, -5];
console.log(mergeSort(unsortedArr));

# 堆排序

参考： https://levelup.gitconnected.com/heapsort-for-javascript-newbies-598d25477d55

function heapify(arr, len, i) {
  	let larget = i;
  	let left = 2 * i + 1;
  	let right = left + 1;
  	if (left < len && arr[left] > arr[larget]) {
    	larget = left;
    }
  	if (right < len && arr[right] > arr[larget]) {
    	larget = right;
    }
  	
  	if (i != larget) {
    	[arr[i], arr[larget]] = [arr[larget], arr[i]];
      	heapify(arr, len, larget);
    }
	return arr;
}

function heapSort(arr) {
    let len = arr.length;
  	let i = Math.floor(len / 2 - 1), k = len - 1;

  	while(i >= 0) {
    	heapify(arr, len, i);
      	i--;
    }
  	
  	while (k >= 0) {
    	[arr[0], arr[k]] = [arr[k], arr[0]];
      	heapify(arr, k, 0);
      	k--;
    }
  	return arr;
}
var unsortedArr = [13, 9, 89, 54, 67, 6];
console.log(heapSort(unsortedArr));

function shellSort(arr) {
    for (let gap = arr.length / 2; gap > 0; gap /= 2) {
        for (let i = gap; i < arr.length; i++) {
            let j = i;
            while(j - gap >= 0 && arr[j-gap] > arr[j]){
                swap(arr, j, j-gap);
                j -= gap;
            }
        }
    }
    return arr;
}
var unsortedArr = [1, 4, 2, 7, 9, 8, 3, 6];
console.log(shellSort(unsortedArr));
