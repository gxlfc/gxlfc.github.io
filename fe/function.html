let data = [{}]; // 如上所示
let res = [];
let keyword = '我';

function filterArr(arr, res, parents) {
    arr.foreach(item => {
        if (item.title.indexOf(keyword) !== -1) {
            res.push({
                title: item.title,
                parents: [...parents] // 拷贝一份，不然操作parents会对中间结果产生影响
            });
        }
        // 如果该节点包含子节点，证明是一个父节点
        if (Array.isArray(item.child)) {
            parents.push({
                title: item.title
            });
            filterArr(item.child, res, parents);
            // 叶子节点终于遍历结束了
            parents.pop();
        }
    });
}
filterArr(data, res, parents);

# 函数柯里化： curry

一个函数不直接（在一次调用中）接受所需的全部参数。

  • 首先接受一个参数，然后返回一个新函数，
  • 然后需要对这个新函数传入第二个参数并继续执行，然后返回第二个新函数，
  • 继续传入第三个参数，以此类推，直到所有必须的参数都以此传入这个柯里化的函数链中
  • 这个函数链的最后一个函数的返回值就是我们期望的那个值；

常用工具库：lodash、Ramda； TODO 源码解析

// 未柯里化
let familyNumber = (name,idol,hobby) => 
    `introduce a familyNumber, name is ${name}, idol is ${idol}, hobby is ${hobby}`;
// 柯里化
let curryFamilyNumber = 
    name => 
        idol => 
            hobby => 
                `introduce a familyNumber, name is ${name}, idol is ${idol}, hobby is ${hobby}`;
curryFamilyNumber('orchid')('jay')('music'); 
familyNumber('orchid', 'jay', 'music');
// introduce a familyNumber,name is orchid,idol is jay, hobby is music
